# Porcentaje de valores null en cada variable
def calculate_missing_percentage(df):
    cols_missing_val_percentage = {}
    cols_missing_val = get_cols_missing_val(df)
    for col in cols_missing_val:
        percentage = (df[col].isnull().sum())
        dimension = len(df)
        final_percentage = round((percentage/dimension)*100,4)
        cols_missing_val_percentage[col] = final_percentage
    return  cols_missing_val_percentage

De acuerdo con el datacard de la API, los atributos country y artistNationality responden de diferente forma a una misma pregunta: Pas de origen de la obra o el artista. El anlisis exploratorio de los datos mostr que, en muchos casos, los valores faltantes de country estaban en artistNationality, y viceversa.

Con el objetivo de unificar la informacin y, por ende, enriquecerla, se construy una clase NationalityMapper compuesta por dos diccionarios, que permiti mapear las nacionalidades a los pases y, a su vez, normalizar los nombres de los pases

Debido a que una obra puede pertenecer a varios artistas, y cada artista puede tener una nacionalidad distinta, se decide que la obra ser de la nacionalidad predominante de los artistas involucrados. Si no existe una nacionalidad predominante, se imputa como 'Otro'

# Funcin que extrae la nacionalidad de cada artista.
def extract_first_nationality(artist_nationalities):
    artist_nationalities_list = artist_nationalities.split("\|")
    first_nationalities = []
    for nationality in artist_nationalities_list:
        first_nationality = nationality.replace(",", " ").split(" ")[0].strip()
        first_nationalities.append(first_nationality)
    return "|".join(first_nationalities)

# Funcin que toma la lista de pases y busca expresiones regulares para obtener los datos que pueden considerarse irregulares o incosistentes.
def get_inconsistent_countries(country_values):
    # Patrn de expresiones regulares para encontrar delimitadores no estndares
    non_standard_delimiters_pattern = r'[;:?]'
    # Lista de descripciones ambiguas
    ambiguous_descriptions = ['possibly', 'probably', 'etc', 'present-day', 'present day']
    # Patrn de expresiones regulares para encontrar palabras completas
    ambiguous_words_pattern = r'\b(?:' + '|'.join(ambiguous_descriptions) + r')\b'
    inconsistencies = []

    for value in country_values:
        # Busca los delimitadores no estndares
        if re.search(non_standard_delimiters_pattern, value):
            inconsistencies.append(value)
        # Busca descripciones ambiguas
        elif re.search(ambiguous_words_pattern, value.lower()):
            inconsistencies.append(value)
    return inconsistencies

def standardize_value(value, known_mappings):
